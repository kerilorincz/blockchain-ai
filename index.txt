    let marketChart;

    function createMarketChart() {
        const ctx = document.getElementById('marketChart').getContext('2d');
        marketChart = new Chart(ctx, {
            type: 'line',
            data: {
                labels: [],
                datasets: []
            },
            options: {
                responsive: true,
                maintainAspectRatio: false,
                plugins: {
                    legend: {
                        position: 'top',
                    },
                    tooltip: {
                        callbacks: {
                            label: function(context) {
                                let label = context.dataset.label || '';
                                if (label) {
                                    label += ': ';
                                }
                                if (context.parsed.y !== null) {
                                    label += new Intl.NumberFormat('en-US', { style: 'currency', currency: 'USD', notation: 'compact', maximumFractionDigits: 2 }).format(context.parsed.y * 1000000);
                                }
                                return label;
                            }
                        }
                    }
                },
                scales: {
                    y: {
                        beginAtZero: true,
                        title: {
                            display: true,
                            text: 'Market Size (USD)'
                        },
                        ticks: {
                           callback: function(value, index, ticks) {
                               return new Intl.NumberFormat('en-US', { notation: 'compact', compactDisplay: 'short' }).format(value * 1000000);
                           }
                        }
                    },
                    x: {
                        title: {
                            display: true,
                            text: 'Year'
                        }
                    }
                }
            }
        });
        setupMarketChartControls();
        updateMarketChart(Object.keys(appData.marketForecasts));
    }

    function setupMarketChartControls() {
        const controlsContainer = document.getElementById('market-chart-controls');
        Object.keys(appData.marketForecasts).forEach(key => {
            const button = document.createElement('button');
            button.textContent = key;
            button.dataset.key = key;
            button.className = 'tab-btn px-3 py-1.5 text-xs font-medium rounded-full border';
            controlsContainer.appendChild(button);
        });

        controlsContainer.addEventListener('click', (e) => {
            if (e.target.tagName === 'BUTTON') {
                e.target.classList.toggle('active');
                e.target.classList.toggle('bg-teal-500');
                e.target.classList.toggle('text-white');
                e.target.classList.toggle('border-teal-500');
                
                const activeKeys = Array.from(controlsContainer.querySelectorAll('button.active')).map(b => b.dataset.key);
                updateMarketChart(activeKeys);
            }
        });
        
        // Activate all by default
        controlsContainer.querySelectorAll('button').forEach(b => {
            b.classList.add('active', 'bg-teal-500', 'text-white', 'border-teal-500');
        });
    }

    function updateMarketChart(activeKeys) {
        const allYears = new Set();
        activeKeys.forEach(key => {
            appData.marketForecasts[key].years.forEach(year => allYears.add(year));
        });
        const sortedYears = Array.from(allYears).sort((a, b) => a - b);

        marketChart.data.labels = sortedYears;
        marketChart.data.datasets = activeKeys.map(key => {
            const forecast = appData.marketForecasts[key];
            const dataPoints = sortedYears.map(year => {
                const index = forecast.years.indexOf(year);
                return index !== -1 ? forecast.values[index] : null;
            });

            return {
                label: `${key} (CAGR: ${forecast.cagr}%)`,
                data: dataPoints,
                fill: false,
                borderColor: forecast.borderColor,
                backgroundColor: forecast.color,
                tension: 0.1,
                spanGaps: true
            };
        });
        marketChart.update();
    }

    function setupModal() {
        const modal = document.getElementById('company-modal');
        const closeBtn = document.getElementById('modal-close-btn');
        const modalTitle = document.getElementById('modal-title');
        const modalCategory = document.getElementById('modal-category');
        const modalBody = document.getElementById('modal-body');
        const modalLinks = document.getElementById('modal-links'); // New element for links

        document.body.addEventListener('click', e => {
            const card = e.target.closest('.company-card');
            if (card) {
                const companyData = JSON.parse(card.dataset.company);
                const category = card.dataset.category;
                
                modalTitle.textContent = companyData.name;
                modalCategory.textContent = category;
                
                let bodyHTML = '';
                // Add Headquarters
                if (companyData.headquarters) {
                    bodyHTML += `<p><strong class="font-semibold text-zinc-700">Headquarters:</strong> ${companyData.headquarters}</p>`;
                }
                // Add CEO
                if (companyData.ceo && companyData.ceo.name) {
                    bodyHTML += `<p><strong class="font-semibold text-zinc-700">CEO:</strong> `;
                    if (companyData.ceo.bioLink) {
                        bodyHTML += `<a href="${companyData.ceo.bioLink}" target="_blank" class="text-teal-600 hover:underline">${companyData.ceo.name}</a>`;
                    } else {
                        bodyHTML += companyData.ceo.name;
                    }
                    bodyHTML += `</p>`;
                }

                for (const [key, value] of Object.entries(companyData)) {
                    // Exclude 'recentNews' from the general loop
                    if (key !== 'name' && key !== 'website' && key !== 'stockSymbol' && key !== 'headquarters' && key !== 'ceo' && key !== 'recentNews') { 
                        bodyHTML += `<p><strong class="font-semibold text-zinc-700 capitalize">${key.replace(/([A-Z])/g, ' $1')}:</strong> ${value}</p>`;
                    }
                }
                modalBody.innerHTML = bodyHTML;

                let linksHTML = '';
                if (companyData.website) {
                    linksHTML += `<a href="${companyData.website}" target="_blank" class="text-teal-600 hover:underline flex items-center"><span class="mr-2">ðŸ”—</span> Company Website</a>`;
                }
                if (companyData.stockSymbol) {
                    const googleFinanceUrl = `https://www.google.com/finance/quote/${companyData.stockSymbol}:NASDAQ`; // Assuming NASDAQ for simplicity
                    linksHTML += `<a href="${googleFinanceUrl}" target="_blank" class="text-teal-600 hover:underline flex items-center"><span class="mr-2">ðŸ“ˆ</span> View Stock on Google Finance</a>`;
                }

                // Removed Recent News section as requested.

                if (companyData.website || companyData.stockSymbol) { // Update condition to exclude recentNews
                    linksHTML += `<p class="text-xs text-zinc-500 mt-2">Note: Live stock data and comprehensive, real-time analyst reports are not available directly within this application due to API limitations and data licensing.</p>`;
                }

                modalLinks.innerHTML = linksHTML; // Populate the new links container

                modal.classList.remove('opacity-0', 'pointer-events-none');
                modal.querySelector('.modal-content').classList.remove('scale-95');
                modal.querySelector('.modal-content').classList.add('scale-100');
            }
        });

        const closeModal = () => {
            modal.classList.add('opacity-0', 'pointer-events-none');
            modal.querySelector('.modal-content').classList.add('scale-95');
            modal.querySelector('.modal-content').classList.remove('scale-100');
        };

        closeBtn.addEventListener('click', closeModal);
        modal.addEventListener('click', e => {
            if (e.target === modal) {
                closeModal();
            }
        });
    }

    createMarketChart();
    populateEcosystem();
    populateApplications();
    populateStartups(); // Call the new function to populate startups
    populateOutlook();
    setupModal();
});
